### Task 1.1: Fix SSRF DNS Rebinding (#10)

**Files:**
- Modify: `html_fetcher.py:110-122` (validate_url DNS resolution)
- Test: `tests/test_security.py`

The current `validate_url` resolves DNS separately from the HTTP request. An attacker can serve a DNS record that resolves to a public IP on first lookup (passing validation) then to 169.254.169.254 on the second lookup (used by `requests.get`). Fix: pin the resolved IP and use it for the actual request.

- [ ] **Step 1: Write failing test for DNS pinning**

In `tests/test_security.py`, add:

```python
class TestDNSPinning:
    """validate_url must pin the resolved IP for use by the HTTP client."""

    def test_dns_rebinding_blocked(self):
        """If DNS resolves to public IP first, then private IP, request must use the pinned public IP."""
        from html_fetcher import HTMLFetcher
        resolve_results = [
            [(None, None, None, None, ("1.2.3.4", 0))],      # First call: public
            [(None, None, None, None, ("169.254.169.254", 0))], # Second call: private (rebinding)
```

```python
@staticmethod
def validate_url_with_pin(url):
    """Validate URL for SSRF and return (is_safe, resolved_ip).
    The resolved IP should be used for the actual HTTP request to prevent DNS rebinding."""
    try:
        parsed = urlparse(url)
    except Exception:
        return False, None

    if parsed.scheme not in ('http', 'https'):
        return False, None

    hostname = parsed.hostname
    if not hostname:
        return False, None

    if hostname in ('169.254.169.254', 'metadata.google.internal'):
        return False, None

    try:
        resolved_ip = socket.getaddrinfo(hostname, None)[0][4][0]
        ip = ipaddress.ip_address(resolved_ip)
        if ip.is_private or ip.is_reserved or ip.is_loopback or ip.is_link_local:
            logging.warning("Rejected URL resolving to private/reserved IP: %s -> [redacted]", url)
            return False, None
    except (socket.gaierror, ValueError):
        logging.warning(f"Could not resolve hostname for URL: {url}")
        return False, None

    return True, resolved_ip
```

Then update `fetch_and_save_if_changed` and `fetch_html` to use the pinned IP.

In `database/snapshots.py`, merge the hash check into the transaction:

```python
def append_researcher_snapshot(researcher_id, position, affiliation, description, source_url=None):
    content_hash = _compute_researcher_content_hash(position, affiliation, description)
    now = datetime.now(timezone.utc)

    with get_connection() as conn:
        with conn.cursor(dictionary=True) as cursor:
            # Check hash inside the same transaction
            cursor.execute(
                "SELECT content_hash FROM researcher_snapshots "
                "WHERE researcher_id = %s ORDER BY scraped_at DESC LIMIT 1",

            )

            cursor.execute(
                """INSERT INTO researcher_snapshots
                   (researcher_id, position, affiliation, description, scraped_at, source_url, content_hash)
                   VALUES (%s, %s, %s, %s, %s, %s, %s)""",
                (researcher_id, position, affiliation, description, now, source_url, content_hash),
            )
            cursor.execute(
                """UPDATE researchers
                   SET position = %s, affiliation = %s, description = %s, description_updated_at = %s
                   WHERE id = %s""",
```

```python
class HTMLFetcher:
    # ... existing class vars ...
    _robots_cache: dict[str, RobotFileParser | None] = {}

    @staticmethod
    def _get_robots_parser(url) -> RobotFileParser | None:
        """Get or fetch the RobotFileParser for a URL's domain. Cached per domain."""
        parsed = urlparse(url)
        origin = f"{parsed.scheme}://{parsed.netloc}"
        if origin in HTMLFetcher._robots_cache:
            return HTMLFetcher._robots_cache[origin]

        try:
            robots_url = f"{origin}/robots.txt"
            resp = requests.get(robots_url, timeout=10, headers={
                'User-Agent': SCRAPER_USER_AGENT,
            })
            if resp.status_code != 200:
                HTMLFetcher._robots_cache[origin] = None
                return None
            rp = RobotFileParser()
            rp.parse(resp.text.splitlines())
            HTMLFetcher._robots_cache[origin] = rp
            return rp
        except Exception:
            HTMLFetcher._robots_cache[origin] = None
            return None

    @staticmethod
    def is_allowed_by_robots(url):
        rp = HTMLFetcher._get_robots_parser(url)
        if rp is None:
            return True
        allowed = rp.can_fetch('HTMLFetcher/1.0', url)
        if not allowed:
            logging.info(f"URL disallowed by robots.txt: {url}")
        return allowed
```
